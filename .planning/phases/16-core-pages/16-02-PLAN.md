The page manages several pieces of local state:
- `search`: string for filtering
- `createOpen`: boolean for create Dialog
- `newKeyName`: string input in create Dialog
- `createdKey`: string | null — the raw key shown after creation
- `revokeTarget`: ApiKey | null — which key the AlertDialog is confirming revoke for
- `copied`: boolean — for copy animation

Structure:

```tsx
import { useState } from 'react';
import { Key, Plus, Copy, Check, Trash2, Search } from 'lucide-react';
import { useKeys, type ApiKey } from '../hooks/useKeys.js';
import { Table, TableHeader, TableBody, TableHead, TableRow, TableCell } from '@/components/ui/table';
import { Dialog, DialogContent, DialogHeader, DialogTitle, DialogDescription, DialogFooter } from '@/components/ui/dialog';
import { AlertDialog, AlertDialogContent, AlertDialogHeader, AlertDialogTitle, AlertDialogDescription, AlertDialogFooter, AlertDialogAction, AlertDialogCancel } from '@/components/ui/alert-dialog';
import { Badge } from '@/components/ui/badge';
import { Button } from '@/components/ui/button';
import { Input } from '@/components/ui/input';

export function KeysPage() {
  const { keys, loading, error, createKey, revokeKey } = useKeys();
  const [search, setSearch] = useState('');
  const [createOpen, setCreateOpen] = useState(false);
  const [newKeyName, setNewKeyName] = useState('');
  const [createdKey, setCreatedKey] = useState<string | null>(null);
  const [creating, setCreating] = useState(false);
  const [revokeTarget, setRevokeTarget] = useState<ApiKey | null>(null);
  const [copied, setCopied] = useState(false);

  const filtered = keys.filter(k =>
    k.key_prefix.toLowerCase().includes(search.toLowerCase()) ||
    k.name.toLowerCase().includes(search.toLowerCase())
  );

  const handleCreate = async () => {
    setCreating(true);
    const raw = await createKey(newKeyName || undefined);
    setCreating(false);
    if (raw) {
      setCreatedKey(raw);
      // Don't close dialog — show the key
    }
  };

  const handleCloseCreate = () => {
    setCreateOpen(false);
    setCreatedKey(null);
    setNewKeyName('');
  };

  const handleCopy = async (text: string) => {
    await navigator.clipboard.writeText(text);
    setCopied(true);
    setTimeout(() => setCopied(false), 2000);
  };

  const handleRevoke = async () => {
    if (revokeTarget) {
      await revokeKey(revokeTarget.id);
      setRevokeTarget(null);
    }
  };

  const formatDate = (iso: string) =>
    new Date(iso).toLocaleDateString('en-US', { month: 'short', day: 'numeric', year: 'numeric' });

  // --- RENDER ---

  // Page header with search + create button
  // If loading: "Loading keys..."
  // If keys.length === 0 && !loading: empty state card
  // Else: search bar + table

  // Table rows map over `filtered`, each row has:
  //   - key_prefix in monospace font
  //   - name
  //   - tier as Badge (variant="outline")
  //   - formatDate(created_at)
  //   - Status: Badge green "Active" or red "Revoked"
  //   - Actions: Copy prefix button (ghost, icon) + Revoke button (ghost, destructive icon) — only for active keys

  // Create Dialog:
  //   - If !createdKey: show name Input + "Create Key" Button
  //   - If createdKey: show success message + raw key in bg-muted rounded p-3 font-mono text-sm + Copy button

  // AlertDialog for revoke:
  //   - Title: "Revoke API Key"
  //   - Description: "Are you sure you want to revoke {key_prefix}? This action cannot be undone."
  //   - Cancel + "Revoke" (destructive variant) buttons

  return (/* full JSX as described above */);
}
```
