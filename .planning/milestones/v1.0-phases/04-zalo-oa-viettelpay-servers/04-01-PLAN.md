export function register(server: McpServer): void {
  server.tool('tool_name', 'Description', { /* zod schema */ }, async (args) => {
    try {
      const result = await client.method(args);
      return { content: [{ type: 'text' as const, text: JSON.stringify(result) }] };

  });
}
